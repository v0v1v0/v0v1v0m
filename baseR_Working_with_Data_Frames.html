  Score = c(89.5, 92.0, 76.5)

df

matrix_df

## Section 2: Numeric-Only Data Frame to Matrix Conversion

  Weight = c(150, 180, 175)
)
df_numeric

## Convert to a matrix

matrix_numeric

## Section 3: Converting a Matrix to a Data Frame
## Initialize a matrix
my_matrix <- matrix(1:9, nrow = 3)
my_matrix

## Convert to a data frame
df_from_matrix <- as.data.frame(my_matrix)
df_from_matrix

## Section 4: Adding Column Names for Clarity
## Renaming columns
colnames(df_from_matrix) <- c("X1", "X2", "X3")

df_from_matrix
