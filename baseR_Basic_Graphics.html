## Example 2: Adding and Customizing Axis Labels

     main = "X vs. X Squared",

graphics::mtext

## Adding Legends
## Example 3: Adding a Legend

## Data for a scatter plot with two groups
x <- 1:10
y1 <- x^2
y2 <- x^1.5

## Plotting two lines on the same plot
plot(x, y1, 
     type = "l", col = "blue", 
     lwd = 2, ylim = c(0, max(y1, y2)),
     main = "Comparing Two Functions", 
     xlab = "X", ylab = "Y")
lines(x, y2, 
      col = "red", lwd = 2)

## Adding a legend
